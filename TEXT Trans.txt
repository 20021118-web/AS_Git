$ROOT="C:\ServerBox"; $nssm="$ROOT\install\nssm.exe"; $caddy="$ROOT\gateway\caddy.exe"
New-Item -ItemType Directory -Force "$ROOT\logs","$ROOT\gateway" | Out-Null
if (-not (Test-Path "$ROOT\gateway\Caddyfile")) {
@'
:80 {
	handle_path /web/* { root * ../apps/webapp
		file_server }
	redir /web /web/ 308
	handle_path /parts/* { reverse_proxy 127.0.0.1:8001 }
	redir /parts /parts/ 308
	handle { redir * /web/ 308 }
}
'@ | Set-Content "$ROOT\gateway\Caddyfile" -Encoding UTF8 }
if (Get-Service ServerBox-Gateway -ErrorAction SilentlyContinue) {
  & $nssm stop ServerBox-Gateway 2>$null | Out-Null
  & $nssm remove ServerBox-Gateway confirm 2>$null | Out-Null; Start-Sleep 1 }
& $nssm install ServerBox-Gateway $caddy "run --config Caddyfile" 2>$null | Out-Null
& $nssm set ServerBox-Gateway AppDirectory "$ROOT\gateway" 2>$null | Out-Null
& $nssm set ServerBox-Gateway AppStdout "$ROOT\logs\gateway.log" 2>$null | Out-Null
& $nssm set ServerBox-Gateway AppStderr "$ROOT\logs\gateway.err.log" 2>$null | Out-Null
& $nssm set ServerBox-Gateway Start SERVICE_AUTO_START 2>$null | Out-Null
& $nssm start ServerBox-Gateway 2>$null | Out-Null
New-NetFirewallRule -DisplayName "ServerBox HTTP" -Direction Inbound -Protocol TCP -LocalPort 80 -Action Allow -Profile Any -ErrorAction SilentlyContinue | Out-Null
Start-Sleep 2; Get-Service ServerBox-Gateway
