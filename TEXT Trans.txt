"=== 1. 서비스 상태 ==="; Get-Service ServerBox-Gateway | Format-List Name,Status
"=== 2. 80포트 점유자 ==="
$c = Get-NetTCPConnection -LocalPort 80 -State Listen -ErrorAction SilentlyContinue
if ($c) { $c | ForEach-Object { (Get-Process -Id $_.OwningProcess).ProcessName + "  (PID " + $_.OwningProcess + ")" } } else { "(80포트를 듣는 프로세스 없음 = Caddy 미기동)" }
"=== 3. Caddy 에러로그 ==="; Get-Content C:\ServerBox\logs\gateway.err.log -Tail 20 -ErrorAction SilentlyContinue
"=== 4. localhost 테스트 ==="; try { (Invoke-WebRequest http://localhost/web/ -UseBasicParsing -TimeoutSec 5).StatusCode } catch { $_.Exception.Message }
